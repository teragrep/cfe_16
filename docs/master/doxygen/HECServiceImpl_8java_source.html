185 @Override
186 public JsonNode getAcks(HttpServletRequest request, String channel, JsonNode requestedAcksInJson) {
187
188 // filter out error cases
189 // authentication header is required always
190 if (this.tokenManager.tokenIsMissing(request)) {
191 throw new AuthenticationTokenMissingException("Authentication token must be provided");
192 }
193
194 String authHeader = request.getHeader("Authorization");
195
196 String authToken;
197 if (tokenManager.isTokenInBasic(authHeader)) {
198 LOGGER.debug("Token was provided via Basic");
199 authToken = this.tokenManager.getTokenFromBasic(authHeader);
200 }
201 else {
202 LOGGER.debug("Token was provided via header");
203 authToken = authHeader;
204 }
205
206 // channel is required
207 if (channel == null) {
208 throw new ChannelNotProvidedException("Channel must be provided when requesting ack statuses");
209 }
210
211 // session is also required
212 Session session = this.sessionManager.getSession(authToken);
213 if (session == null) {
214 throw new SessionNotFoundException("Session not found for auth token " + authToken);
215 }
216
217 // if channel is not inside Session, it is considered an error case
218 if (!session.doesChannelExist(channel)) {
219 throw new ChannelNotFoundException();
220 }
221 session.touch();
222
223 ObjectNode responseNode = objectMapper.createObjectNode();
224 JsonNode requestedAckStatuses = this.acknowledgements
225 .getRequestedAckStatuses(authToken, channel, requestedAcksInJson);
226 responseNode.put("acks", requestedAckStatuses);
227 return responseNode;
228 }
