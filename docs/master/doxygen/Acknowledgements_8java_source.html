202 /*
203 * Assignes an Ack value for the event. Checks it there are acks still available
204 * for the channel If there are no Acks available, throws ServerIsBusyException.
205 * Uses the ackToCompare variable to check if Ack with a current value is in the
206 * Ack list and increases the Ack value until a suitable Ack value is found and
207 * when it is found, a new Ack with the current Ack value as an id is created.
208 */
209 public boolean incrementAckValue(String authToken, String channel) {
210 String key = authToken + channel;
211 State state = this.ackStates.get(key);
212 if (state == null) {
213 return false;
214 }
215
216 if (!acksAvailable(state)) {
217 throw new ServerIsBusyException();
218 }
219
220 int currentAckValue;
221 synchronized (state) {
222 currentAckValue = state.getCurrentAckValue();
223 Ack ackToCompare = state.getAckToCompare();
224 ackToCompare.setId(currentAckValue);
225 state.setAckToCompare(ackToCompare);
226 Map<Integer, Ack> ackMap = state.getAckMap();
227 while (ackMap.containsKey(ackToCompare.getId())) {
228 currentAckValue++;
229 ackToCompare = state.getAckToCompare();
230 ackToCompare.setId(currentAckValue);
231 if (currentAckValue > this.configuration.maxAckValue()) {
232 currentAckValue = 0;
233 }
234 state.setAckToCompare(ackToCompare);
235 }
236
237 currentAckValue++;
238 if (currentAckValue > this.configuration.maxAckValue()) {
239 currentAckValue = 0;
240 }
241 state.setCurrentAckValue(currentAckValue);
242 }
243 return true;
244 }

246 /*
247 * Uses the help Ack object to find the Ack object from the Ack set and sets the
248 * acknowledgement status of the Ack as true.
249 */
250 public boolean acknowledge(String authToken, String channel, int ackId) {
251 String key = authToken + channel;
252 State state = this.ackStates.get(key);
253 LOGGER.debug("Acknowledging ackId <{}> on channel <{}>", ackId, channel);
254 if (state == null) {
255 throw new IllegalStateException("An Ack cannot be acknowledge before it is added to the Ack list.");
256 }
257 synchronized (state) {
258 Map<Integer, Ack> ackMap = state.getAckMap();
259 Ack ack = ackMap.get(ackId);
260 if (ack == null) {
261 throw new InternalServerErrorException("Couldn't set the acknowledge status for Ack ID " + ackId);
262 }
263 ack.acknowledge();
264 return true;
265 }
266 }

311 public JsonNode getRequestedAckStatuses(String authToken, String channel, JsonNode requestedAcksInJson) {
312 JsonNode jsonNode = this.objectMapper.createObjectNode();
313 if (requestedAcksInJson == null) {
314 return jsonNode;
315 }
316
317 /*
318 * Checks that the JSON parameter is given, and that there is an Ack node which
319 * is an array. Saves the requested ack ids in an int array.
320 */
321 int[] requestedAckIds = null;
322 if (requestedAcksInJson.get("acks") != null && requestedAcksInJson.get("acks").isArray()) {
323 requestedAckIds = this.objectMapper.convertValue(requestedAcksInJson.get("acks"), int[].class);
324 }
325
326 /*
327 * Goes through the requested Ack id:s, checks if they have been assigned to an
328 * Ack object in the Ack list. If an Ack is found, the Ack id and the
329 * acknowledgement status of that Ack is saved in "ackStatuses" JSON node. If
330 * Ack is not found in the list, the Ack id and acknowledgement status "false"
331 * is added to the JSON node.
332 */
333
334 Map<Integer, Boolean> ackStatuses = new HashMap<Integer, Boolean>();
335 if (requestedAckIds != null) {
336 String key = authToken + channel;
337 State state = this.ackStates.get(key);
338 if (state == null) {
339 return null;
340 }
341 synchronized (state) {
342 Map<Integer, Ack> ackMap = state.getAckMap();
343 for (int i = 0; i < requestedAckIds.length; i++) {
344 int ackId = requestedAckIds[i];
345 Ack ack = ackMap.get(ackId);
346 if (ack == null) {
347 ackStatuses.put(ackId, false);
348 }
349 else {
350 ackStatuses.put(ackId, ack.isAcknowledged());
351 ackMap.remove(ackId);
352 }
353 }
354 }
355 }
356 jsonNode = this.objectMapper.convertValue(ackStatuses, JsonNode.class);
357 return jsonNode;
358 }

360 /*
361 * Checks if there is room in the Ack list for a new entry
362 *
363 * Not thread safe, needs external synchronization.
364 */
365 private boolean acksAvailable(State state) {
366 Map<Integer, Ack> ackMap = state.getAckMap();
367 int ackMapSize = ackMap.size();
368 int maxAckValue = this.configuration.maxAckValue();
369 if (ackMapSize > maxAckValue) {
370 return false;
371 }
372 else {
373 return true;
374 }
375 }

382 public boolean deleteAckFromList(String authToken, String channel, Ack ack) {
383 String key = authToken + channel;
384 State state = this.ackStates.get(key);
385 if (state == null) {
386 throw new InternalServerErrorException("No State for key " + key);

394 public void run() {
395
396 while (true) {
397 try {
398 LOGGER.debug("Sleeping for <{}> while waiting for polls", this.configuration.pollTime());
399 Thread.sleep(this.configuration.pollTime());
400 }
401 catch (InterruptedException e) {
402 break;
403 }
404
405 for (String key : this.ackStates.keySet()) {
406 State state = this.ackStates.get(key);
407
408 synchronized (state) {
409 Map<Integer, Ack> ackMap = state.getAckMap();
410 Iterator<Ack> iterator = ackMap.values().iterator();
411 while (iterator.hasNext()) {
412 Ack ack = iterator.next();
413 long thresholdInLong = ack.getLastUsedTimestamp() + this.configuration.maxAckAge();
414
418 long now = System.currentTimeMillis();
419 if (now >= thresholdInLong) {
420 iterator.remove();
421 }
422 }
423 }
424 }
425 }
