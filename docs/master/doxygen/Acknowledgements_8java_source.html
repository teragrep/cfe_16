196 /*
197 * Assignes an Ack value for the event. Checks it there are acks still available
198 * for the channel If there are no Acks available, throws ServerIsBusyException.
199 * Uses the ackToCompare variable to check if Ack with a current value is in the
200 * Ack list and increases the Ack value until a suitable Ack value is found and
201 * when it is found, a new Ack with the current Ack value as an id is created.
202 */
203 public boolean incrementAckValue(String authToken, String channel) {
204 String key = authToken + channel;
205 State state = this.ackStates.get(key);
206 if (state == null) {
207 return false;
208 }
209
210 if (!acksAvailable(state)) {
211 throw new ServerIsBusyException();
212 }
213
214 int currentAckValue;
215 synchronized (state) {
216 currentAckValue = state.getCurrentAckValue();
217 Ack ackToCompare = state.getAckToCompare();
218 ackToCompare.setId(currentAckValue);
219 state.setAckToCompare(ackToCompare);
220 Map<Integer, Ack> ackMap = state.getAckMap();
221 while (ackMap.containsKey(ackToCompare.getId())) {
222 currentAckValue++;
223 ackToCompare = state.getAckToCompare();
224 ackToCompare.setId(currentAckValue);
225 if (currentAckValue > this.configuration.getMaxAckValue()) {
226 currentAckValue = 0;
227 }
228 state.setAckToCompare(ackToCompare);
229 }
230
231 currentAckValue++;
232 if (currentAckValue > this.configuration.getMaxAckValue()) {
233 currentAckValue = 0;
234 }
235 state.setCurrentAckValue(currentAckValue);
236 }
237 return true;
238 }

240 /*
241 * Uses the help Ack object to find the Ack object from the Ack set and sets the
242 * acknowledgement status of the Ack as true.
243 */
244 public boolean acknowledge(String authToken, String channel, int ackId) {
245 String key = authToken + channel;
246 State state = this.ackStates.get(key);
247 LOGGER.debug("Acknowledging ackId <{}> on channel <{}>", ackId, channel);
248 if (state == null) {
249 throw new IllegalStateException("An Ack cannot be acknowledge before it is added to the Ack list.");
250 }
251 synchronized (state) {
252 Map<Integer, Ack> ackMap = state.getAckMap();
253 Ack ack = ackMap.get(ackId);
254 if (ack == null) {
255 throw new InternalServerErrorException("Couldn't set the acknowledge status for Ack ID " + ackId);
256 }
257 ack.acknowledge();
258 return true;
259 }
260 }

305 public JsonNode getRequestedAckStatuses(String authToken, String channel, JsonNode requestedAcksInJson) {
306 JsonNode jsonNode = this.objectMapper.createObjectNode();
307 if (requestedAcksInJson == null) {
308 return jsonNode;
309 }
310
311 /*
312 * Checks that the JSON parameter is given, and that there is an Ack node which
313 * is an array. Saves the requested ack ids in an int array.
314 */
315 int[] requestedAckIds = null;
316 if (requestedAcksInJson.get("acks") != null && requestedAcksInJson.get("acks").isArray()) {
317 requestedAckIds = this.objectMapper.convertValue(requestedAcksInJson.get("acks"), int[].class);
318 }
319
320 /*
321 * Goes through the requested Ack id:s, checks if they have been assigned to an
322 * Ack object in the Ack list. If an Ack is found, the Ack id and the
323 * acknowledgement status of that Ack is saved in "ackStatuses" JSON node. If
324 * Ack is not found in the list, the Ack id and acknowledgement status "false"
325 * is added to the JSON node.
326 */
327
328 Map<Integer, Boolean> ackStatuses = new HashMap<Integer, Boolean>();
329 if (requestedAckIds != null) {
330 String key = authToken + channel;
331 State state = this.ackStates.get(key);
332 if (state == null) {
333 return null;
334 }
335 synchronized (state) {
336 Map<Integer, Ack> ackMap = state.getAckMap();
337 for (int i = 0; i < requestedAckIds.length; i++) {
338 int ackId = requestedAckIds[i];
339 Ack ack = ackMap.get(ackId);
340 if (ack == null) {
341 ackStatuses.put(ackId, false);
342 }
343 else {
344 ackStatuses.put(ackId, ack.isAcknowledged());
345 ackMap.remove(ackId);
346 }
347 }
348 }
349 }
350 jsonNode = this.objectMapper.convertValue(ackStatuses, JsonNode.class);
351 return jsonNode;
352 }

354 /*
355 * Checks if there is room in the Ack list for a new entry
356 *
357 * Not thread safe, needs external synchronization.
358 */
359 private boolean acksAvailable(State state) {
360 Map<Integer, Ack> ackMap = state.getAckMap();
361 int ackMapSize = ackMap.size();
362 int maxAckValue = this.configuration.getMaxAckValue();
363 if (ackMapSize > maxAckValue) {
364 return false;
365 }
366 else {
367 return true;
368 }
369 }

376 public boolean deleteAckFromList(String authToken, String channel, Ack ack) {
377 String key = authToken + channel;
378 State state = this.ackStates.get(key);
379 if (state == null) {
380 throw new InternalServerErrorException("No State for key " + key);
381 }
382 synchronized (state) {
383 state.getAckMap().remove(ack.getId());

388 public void run() {
389
390 while (true) {
391 try {
392 LOGGER.debug("Sleeping for <{}> while waiting for polls", this.configuration.getPollTime());
393 Thread.sleep(this.configuration.getPollTime());
394 }
395 catch (InterruptedException e) {
396 break;
397 }
398
399 for (String key : this.ackStates.keySet()) {
400 State state = this.ackStates.get(key);
401
402 synchronized (state) {
403 Map<Integer, Ack> ackMap = state.getAckMap();
404 Iterator<Ack> iterator = ackMap.values().iterator();
405 while (iterator.hasNext()) {
406 Ack ack = iterator.next();
407 long thresholdInLong = ack.getLastUsedTimestamp() + this.configuration.getMaxAckAge();
408
412 long now = System.currentTimeMillis();
413 if (now >= thresholdInLong) {
414 iterator.remove();
415 }
416 }
417 }
418 }
419 }
