69 public RelpConnection(String hostname, int port) {
70 super(hostname, port);

78 synchronized private void connect() {
79 boolean notConnected = true;
80 while (notConnected) {
81 boolean connected = false;
82 try {
83 LOGGER.debug("Connecting to RELP server");
84 connected = this.connection.connect(this.hostname, this.port);
85 }
86 catch (Exception e) {
87 LOGGER.warn("Failed to connect to RELP Server: ", e);
88 }
89 if (connected) {
90 notConnected = false;
91 }
92 else {
93 try {
94 LOGGER.debug("Sleeping for <[{}]> before reconnecting", this.reconnectInterval);
95 Thread.sleep(this.reconnectInterval);
96 }
97 catch (InterruptedException e) {
98 LOGGER.warn("Sleep interrupted: ", e);
99 }
100 }
101 }
102 }
103
104 synchronized private void tearDown() {
105 LOGGER.debug("Tearing down connection");
106 this.connection.tearDown();
107 }
108
109 synchronized private void disconnect() {
110 try {
111 LOGGER.debug("Disconnecting from RELP server");
112 this.connection.disconnect();
113 }
114 catch (IllegalStateException | IOException | TimeoutException e) {
115 LOGGER.warn("Failed to disconnect from RELP Server: ", e);
116 }
117 finally {
118 this.tearDown();
119 }
120 }

127 @Override
128 synchronized public void sendMessages(List<SyslogMessage> syslogMessages) throws IOException {
129 final RelpBatch relpBatch = new RelpBatch();
130 for (SyslogMessage syslogMessage : syslogMessages) {
131 relpBatch.insert(syslogMessage.toRfc5424SyslogMessage().getBytes(StandardCharsets.UTF_8));
132 }
133 doSend(relpBatch);

143 synchronized private void doSend(RelpBatch relpBatch) {
144 boolean notSent = true;
145
146 while (notSent) {
147 try {
148 LOGGER.debug("Committing a RELP batch");
149 this.connection.commit(relpBatch);
150 }
151 catch (IllegalStateException | IOException | TimeoutException e) {
152 LOGGER.warn("Failed to commit batch: ", e);
153 }
154
155 if (!relpBatch.verifyTransactionAll()) {
156 LOGGER.debug("Failed to verify all transactions, retrying them");
157 relpBatch.retryAllFailed();
158 this.tearDown();
159 this.connect();
160 }
161 else {
162 notSent = false;
163 }
164 }
165 }
