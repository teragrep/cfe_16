109 /*
110 * Method used when converting data and the channel is specified in the request.
111 * Takes authentication token, all events sent in a request (in JSON format) and
112 * the channel name as string parameters. Returns a JSON node with ack id if
113 * everything is successful. Example: {"text":"Success","code":0,"ackID":0}
114 */
115 public ObjectNode convertData(
116 String authToken,
117 String channel,
118 String allEventsInJson,
119 HeaderInfo headerInfo,
120 Acknowledgements acknowledgements
121 ) {
122 HttpEventData previousEvent = null;
123
124 acknowledgements.initializeContext(authToken, channel);
125 int ackId = acknowledgements.getCurrentAckValue(authToken, channel);
126 boolean incremented = acknowledgements.incrementAckValue(authToken, channel);
127 if (!incremented) {
128 throw new InternalServerErrorException("Ack value couldn't be incremented.");
129 }
130 Ack ack = new Ack(ackId, false);
131 boolean addedAck = acknowledgements.addAck(authToken, channel, ack);
132 if (!addedAck) {
133 throw new InternalServerErrorException("Ack ID " + ackId + " couldn't be added to the Ack set.");
134 }
135 JsonStreamParser parser = new JsonStreamParser(allEventsInJson);
136
137 /*
138 * There can be multiple events in one request. Here they are handled one by
139 * one. The event is saved in a string variable and is converted into
140 * HttpEventData object. Metadata is assigned to the object. HttpEventData is
141 * converted into SyslogMessage and saved in a list in a RequestInfo object.
142 * After the event is handled, it is assigned as a value to previousEvent
143 * variable.
144 */
145 HttpEventData eventData = null;
146 Converter converter = new Converter(headerInfo);
147 List<SyslogMessage> syslogMessages = new ArrayList<SyslogMessage>();
148 while (parser.hasNext()) {
149 previousEvent = eventData;
150 String jsonObjectStr = parser.next().toString();
151 eventData = verifyJsonData(jsonObjectStr, previousEvent);
152 eventData = assignMetaData(eventData, authToken, channel);
153 SyslogMessage syslogMessage = converter.httpToSyslog(eventData);
154 syslogMessages.add(syslogMessage);
155 }
156
157 /*
158 * SyslogMessage syslogMessage =
159 * converter.getHeaderInfoSyslogMessage(headerInfo);
160 * requestInfo.getConvertedData().add(syslogMessage);
161 */
162 /*
163 * After all the events are sent, previousEvent object is set to null, the
164 * events are sent with the Acknowledgements and ack id and JSON node with an ack id
165 * will be returned informing that the sending of the events has been
166 * successful.
167 */
168 previousEvent = null;
169
170 // create a new object to avoid blocking of threads because
171 // the SyslogMessageSender.sendMessage() is synchronized
172 try {
173 this.connection.sendMessages(syslogMessages);
174 }
175 catch (IOException e) {
176 throw new InternalServerErrorException(e);
177 }
178
179 boolean shouldAck = channel != null && !channel.equals(Session.DEFAULT_CHANNEL);
180
181 if (shouldAck) {
182 boolean acked = acknowledgements.acknowledge(authToken, channel, ackId);
183 if (!acked) {
184 throw new InternalServerErrorException("Ack ID " + ackId + " not Acked.");
185 }
186 }
187
188 ObjectNode responseNode = this.objectMapper.createObjectNode();
189
190 responseNode.put("text", "Success");
191 responseNode.put("code", 0);
192 if (shouldAck) {
193 responseNode.put("ackID", ackId);
194 }
195
196 return responseNode;
197 }

199 /*
200 * Pre-handles the event and assigns it's information into HttpEventData object
201 * and returns it. Information from the string is read with ObjectMapper into a
202 * JsonNode. After that the supposed fields are checked from the JsonNode and if
203 * the field is found, information from it will be saved in HttpEventData
204 * object. Finally handleTime() is called to assign correct time information to
205 * the object. When multiple events are sent in one request, the value of the
206 * fields are saved for the following events. The values can be overriden. If
207 * the value is overridden, it will stay as so for the following events if it is
208 * not overridden.
209 */
210 private HttpEventData verifyJsonData(String eventInJson, HttpEventData previousEvent) {
211
212 HttpEventData eventData = new HttpEventData();
213
214 /*
215 * Event field cannot be missing or blank. Throws an exception if this is the
216 * case.
217 */
218 JsonNode jsonObject;
219 try {
220 jsonObject = this.objectMapper.readTree(eventInJson);
221 }
222 catch (IOException e) {
223 jsonObject = null;
224 }
225
226 if (jsonObject != null) {
227 JsonNode event = jsonObject.get("event");
228 if (event != null) {
229 eventData.setEvent(event.toString());
230 }
231 else {
232 throw new EventFieldMissingException();
233 }
234 if (eventData.getEvent().matches("\"\"") || eventData.getEvent() == null) {
235 throw new EventFieldBlankException();
236 }
237 eventData = handleTime(eventData, jsonObject, previousEvent);
238 }
239 return eventData;
240 }

242 /*
243 * The time stamp of the event can be given as epoch time in the request.
244 */
245 private HttpEventData handleTime(HttpEventData eventData, JsonNode jsonObject, HttpEventData previousEvent) {
246 JsonNode timeObject = jsonObject.get("time");
247
248 /*
249 * If the time is given as a string rather than as a numeral value, the time is
250 * handled in a same way as it is handled when time is not given in a request.
251 */
252 if (timeObject == null || timeObject.isTextual()) {
253 eventData.setTimeParsed(false);
254 eventData.setTimeSource("generated");
255 if (previousEvent != null) {
256 if (previousEvent.isTimeParsed()) {
257 eventData.setTimeAsLong(previousEvent.getTimeAsLong());
258 eventData.setTime(previousEvent.getTime());
259 eventData.setTimeParsed(true);
260 eventData.setTimeSource("reported");
261 }
262 }
263 /*
264 * If the time is given as epoch seconds with a decimal (example:
265 * 1433188255.253), the decimal point must be removed and time is assigned to
266 * HttpEventData object as a long value. convertTimeToEpochMillis() will check
267 * that correct time format is used.
268 */
269 }
270 else if (timeObject.isDouble()) {
271 eventData.setTimeAsLong(removeDecimal(timeObject.asDouble()));
272 eventData.setTime(String.valueOf(eventData.getTimeAsLong()));
273 eventData.setTimeParsed(true);
274 eventData.setTimeSource("reported");
275 eventData = convertTimeToEpochMillis(eventData);
276 /*
277 * If the time is given in a numeral value, it is assigned to HttpEventData
278 * object as a long value. convertTimeToEpochMillis() will check that correct
279 * time format is used.
280 */
281 }
282 else if (timeObject.canConvertToLong()) {
283 eventData.setTimeAsLong(timeObject.asLong());
284 eventData.setTime(jsonObject.get("time").asText());
285 eventData.setTimeParsed(true);
286 eventData.setTimeSource("reported");
287 eventData = convertTimeToEpochMillis(eventData);
288 }
289 else {
290 eventData.setTimeParsed(false);
291 eventData.setTimeSource("generated");
292 }
293
294 return eventData;
295 }

297 /*
298 * Takes a double value as a parameter, removes the decimal point from that
299 * value and returns the number as a long value.
300 */
301 private long removeDecimal(double doubleValue) {
302 BigDecimal doubleValueWithDecimal = BigDecimal.valueOf(doubleValue);
303 String stringValue = doubleValueWithDecimal.toString();
304 String stringValueWithoutDecimal = stringValue.replace(".", "");
305 long longValue = Long.parseLong(stringValueWithoutDecimal);
306
307 return longValue;
308 }

310 /*
311 * Converts the given time stamp into epoch milliseconds. Takes a HttpEventData
312 * object as a parameter. Gets the time from the variable set in the
313 * HttpEventData object. If the time value in the object has 13 digits, it means
314 * that time has been already given in epoch milliseconds.
315 */
316 private HttpEventData convertTimeToEpochMillis(HttpEventData eventData) {
317 String timeString = eventData.getTime();
318 if (timeString.length() == 13) {
319 return eventData;
320 }
321 else if (timeString.length() >= 10 && timeString.length() < 13) {
322 eventData.setTimeAsLong(eventData.getTimeAsLong() * (long) Math.pow(10, ((13 - timeString.length()))));
323 }
324 else {
325 eventData.setTimeParsed(false);
326 eventData.setTimeSource("generated");
327 }
328 return eventData;
329 }
