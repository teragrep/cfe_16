60/*
61 * Manager that handles creating sessions and getting already existing sessions.
62 * Sessions are indexed by the authentication token.
63 *
64 */
65@Component
66public class SessionManager implements Runnable, LifeCycle {
67
68 private static final Logger LOGGER = LoggerFactory.getLogger(SessionManager.class);
72 private final Map<String, Session> sessions;
73
77 private Thread cleanerThread;
78
79 @Autowired
80 private Configuration configuration;
81
85 public SessionManager() {
86 this.sessions = new HashMap<String, Session>();
87 }
88
89 @Override
90 @PostConstruct
91 public void start() {
92 this.cleanerThread = new Thread(this, "Session cleaner");
93 this.cleanerThread.start();
94 }
95
96 @Override
97 public void stop() {
98 this.cleanerThread.interrupt();
99 }
100
101 @Override
102 public void run() {
103 while (true) {
104 try {
105 LOGGER.debug("Sleeping for <{}> while waiting for poll", this.configuration.getPollTime());
106 Thread.sleep(this.configuration.getPollTime());
107 }
108 catch (InterruptedException e) {
109 break;
110 }
111 synchronized (this) {
112 Iterator<Map.Entry<String, Session>> iterator = this.sessions.entrySet().iterator();
113 while (iterator.hasNext()) {
114 Map.Entry<String, Session> entry = iterator.next();
115 long now = System.currentTimeMillis();
116 long thresholdInLong = entry.getValue().getLastTouchedTimestamp()
117 + this.configuration.getMaxSessionAge();
118 if (now >= thresholdInLong) {
119 iterator.remove();
120 }
121 }
122 }
123 }
124 }

126 /*
127 * Gets a session for provided authentication token. returns null if
128 * there is no session for given authentication token
129 */
130 public Session getSession(String authenticationToken) {
131 synchronized (this) {
132 Session session = this.sessions.get(authenticationToken);
133 return session;
134 }
135 }
136
143 public Session getOrCreateSession(String authenticationToken) {
144 LOGGER.debug("Getting or creating session");
145 LOGGER.trace("Getting or creating session for authenticationToken: {}", authenticationToken);
146 synchronized (this) {
147 Session session = this.sessions.get(authenticationToken);
148 if (session == null) {
149 session = new Session(null, authenticationToken);
150 this.sessions.put(authenticationToken, session);
151 }
152 return session;
153 }
154 }
155
156 public void removeSession(String authenticationToken) {
157 LOGGER.debug("Removing session");
158 LOGGER.trace("Removing session for authenticationToken: {}", authenticationToken);
159 synchronized (this) {
160 this.sessions.remove(authenticationToken);
161 }
162 }
163
164 /*
165 * Creates a new session object
166 */
167 public Session createSession(String authenticationToken) {
168 LOGGER.debug("Creating new session");
169 LOGGER.trace("Creating new session for authenticationToken: {}", authenticationToken);
170 synchronized (this) {
171 Session session = new Session(authenticationToken);
172 this.sessions.put(authenticationToken, session);
173 return session;
174 }
