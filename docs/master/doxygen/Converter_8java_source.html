57/*
58 * Converts HTTP Event Data into a Syslog message.
59 *
60 * This class is NOT thread safe!
61 *
62 */
63
64public final class Converter {
65
66 private static final Logger LOGGER = LoggerFactory.getLogger(Converter.class);
67 private Severity severity;
68 private Facility facility;
69
70 private SDElement metadataSDE;
71 private final HeaderInfo headerInfo;
72
73 private final String hostName = "cfe-16";
74
75 public Converter(HeaderInfo headerInfo) {
76 this.headerInfo = headerInfo;
77 }
78
79 public SyslogMessage httpToSyslog(HttpEventData httpEventData) {
80
81 setEventSeverity();
82 setEventFacility();
83
84 setStructuredDataParams(httpEventData);
85
86 SyslogMessage syslogMessage;
87 if (httpEventData.isTimeParsed()) {
88
89 /*
90 * Creates a Syslogmessage with a time stamp
91 */
92 LOGGER.debug("Creating new syslog message with timestamp");
93 syslogMessage = new SyslogMessage()
94 .withTimestamp(httpEventData.getTimeAsLong())
95 .withSeverity(severity)
96 .withAppName("capsulated")
97 .withHostname(hostName)
98 .withFacility(facility)
99 .withSDElement(metadataSDE)
100 .withSDElement(this.headerInfo.asSDElement())
101 .withMsg(httpEventData.getEvent());
102
103 }
104 else {
105 /*
106 * Creates a Syslogmessage without timestamp, because the time is already given
107 * in the request.
108 */
109 LOGGER.debug("Creating new syslog message without timestamp");
110 syslogMessage = new SyslogMessage()
111 .withSeverity(severity)
112 .withAppName("capsulated")
113 .withHostname(hostName)
114 .withFacility(facility)
115 .withSDElement(metadataSDE)
116 .withSDElement(this.headerInfo.asSDElement())
117 .withMsg(httpEventData.getEvent());
118 }
119
120 return syslogMessage;
121 }
122
123 /*
124 * Event severity is coded to always be INFORMATIONAL
125 */
126 private void setEventSeverity() {
127 severity = Severity.INFORMATIONAL;
128 }
129
130 /*
131 * Event facility is coded to always be USER
132 */
133 private void setEventFacility() {
134 facility = Facility.USER;
135 }
136
137 /*
138 * Gets the data from the HTTP Event Data and adds it to SD Element as SD
139 * Parameters.
140 */
141 private void setStructuredDataParams(HttpEventData eventData) {
142 LOGGER.debug("Setting Structured Data params");
143 metadataSDE = new SDElement("cfe_16-metadata@48577");
144
145 if (eventData.getAuthenticationToken() != null) {
146 LOGGER.debug("Setting authentication token");
147 metadataSDE.addSDParam("authentication_token", eventData.getAuthenticationToken());
148 }
149
150 if (eventData.getChannel() != null) {
151 LOGGER.debug("Setting channel");
152 metadataSDE.addSDParam("channel", eventData.getChannel());
153 }
154
155 if (eventData.getAckID() != null) {
156 LOGGER.debug("Setting ack id");
157 metadataSDE.addSDParam("ack_id", String.valueOf(eventData.getAckID()));
158 }
159
160 if (eventData.getTimeSource() != null) {
161 LOGGER.debug("Setting time source");
162 metadataSDE.addSDParam("time_source", eventData.getTimeSource());
163 }
164
165 if (eventData.isTimeParsed()) {
166 LOGGER.debug("Setting time_parsed and time");
167 metadataSDE.addSDParam("time_parsed", "true");
168 metadataSDE.addSDParam("time", eventData.getTime());
169 }
170 }
