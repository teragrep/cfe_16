177 /*
178 * Tests getting the Ack statuses from Acknowledgements. First we create the request
179 * bodies and the supposed responses as strings. Then we create the nodes for
180 * the requests and read the strings into the node object. After that
181 * Acknowledgements' getRequestedAckStatuses() is called with the JsonNode requests
182 * and the response is compared to the supposed responses.
183 */
184 @Test
185 public void getRequestedAckStatusesTest() {
186 String requestAsString = "{\"acks\": [1,3,4]}";
187 String notIntRequestAsString = "{\"acks\": [\"a\",\"b\",\"c\"]}";
188 String faultyRequestAsString = "{\"test\": [1,3,4]}";
189 String supposedResponseAsStringOneTrue = "{\"1\":true,\"3\":false,\"4\":false}";
190 String supposedResponseAsStringAllFalse = "{\"1\":false,\"3\":false,\"4\":false}";
191 ObjectMapper mapper = new ObjectMapper();
192 JsonNode emptyJsonNode = mapper.createObjectNode();
193 JsonNode queryNode = mapper.createObjectNode();
194 JsonNode faultyNode = mapper.createObjectNode();
195 JsonNode notIntNode = mapper.createObjectNode();
196 acknowledgements.initializeContext(this.authToken1, this.channel1);
197 assertTrue(acknowledgements.addAck(this.authToken1, this.channel1, new Ack(1, false)));
198 assertTrue(acknowledgements.acknowledge(this.authToken1, this.channel1, 1));
199 try {
200 queryNode = mapper.readTree(requestAsString);
201 faultyNode = mapper.readTree(faultyRequestAsString);
202 notIntNode = mapper.readTree(notIntRequestAsString);
203 }
204 catch (IOException e) {
205 throw new RuntimeException(e);
206 }
207 assertEquals(
208 "getRequestedAckStatuses should return null, when providing a null value as a parameter", emptyJsonNode,
209 acknowledgements.getRequestedAckStatuses(this.authToken1, "", null)
210 );
211
212 try {
213 assertEquals(
214 "ackId 1 status should be true on channel1, others should be false.",
215 supposedResponseAsStringOneTrue,
216 acknowledgements.getRequestedAckStatuses(this.authToken1, this.channel1, queryNode).toString()
217 );
218 }
219 catch (Throwable e1) {
220 // TODO Auto-generated catch block
221 LOGGER.warn("Failed to handle ack request: ", e1);
222 }
223
224 assertEquals(
225 "ackId 1 status should be false on channel1 after requesting it's status once. All others should be false as well",
226 supposedResponseAsStringAllFalse,
227 acknowledgements.getRequestedAckStatuses(this.authToken1, this.channel1, queryNode).toString()
228 );
229
230 acknowledgements.initializeContext(this.authToken2, this.channel2);
231 assertEquals(
232 "All ack statuses should be false for channel2", supposedResponseAsStringAllFalse,
233 acknowledgements.getRequestedAckStatuses(this.authToken2, this.channel2, queryNode).toString()
234 );
235
236 assertEquals(
237 "An empty JsonNode should be returned when querying with an empty JsonNode", emptyJsonNode,
238 acknowledgements.getRequestedAckStatuses(this.authToken1, this.channel1, emptyJsonNode)
239 );
240
241 assertEquals(
242 "An empty JsonNode should be returned when querying with a JsonNode that has no \"acks\" field in it.",
243 emptyJsonNode, acknowledgements.getRequestedAckStatuses(this.authToken1, this.channel1, faultyNode)
244 );
245 try {
246 assertEquals(
247 "An empty JsonNode should be returned when querying with a JsonNode that has \"acks\" field in it, but it has something else than integers as it's values",
248 emptyJsonNode, acknowledgements.getRequestedAckStatuses(this.authToken1, this.channel1, notIntNode)
249 );
250 fail("Expected IllegalArgumentException");
251 }
252 catch (IllegalArgumentException e) {
253 }
254 }

282 /*
283 * Tests deleting the Ack from Acknowledgements. First we get the list that is
284 * currently in the Acknowledgements and save it to list1 variable. After that we
285 * delete an Ack from the Acknowledgements's list, then we get the list from
286 * Acknowledgements again and save it to list2 variable. These 2 lists are then
287 * compared to each other.
288 */
289 @Test
290 public void deleteAckTest() {
291 Acknowledgements acknowledgements1 = new Acknowledgements();
292 Acknowledgements acknowledgements2 = new Acknowledgements();
293
294 acknowledgements1.initializeContext(this.authToken1, this.channel1);
295 assertTrue(acknowledgements1.addAck(this.authToken1, this.channel1, new Ack(0, false)));
296 assertTrue(acknowledgements1.addAck(this.authToken1, this.channel1, new Ack(1, false)));
297
298 Map<Integer, Ack> list1 = acknowledgements1.getAckList(this.authToken1, this.channel1);
299 int list1Size = acknowledgements1.getAckListSize(this.authToken1, this.channel1);
300 assertEquals("Ack list 1 size should be 2.", 2, list1Size);
301
302 Ack deletedAck = list1.values().iterator().next();
303
304 acknowledgements2.initializeContext(this.authToken2, this.channel2);
305 assertTrue(acknowledgements2.addAck(this.authToken2, this.channel2, new Ack(0, false)));
306
307 acknowledgements2.initializeContext(this.authToken2, this.channel2);
308 assertTrue(acknowledgements2.addAck(this.authToken2, this.channel2, new Ack(1, false)));
309
310 acknowledgements2.deleteAckFromList(this.authToken2, this.channel2, deletedAck);
311 Map<Integer, Ack> list2 = acknowledgements2.getAckList(this.authToken2, this.channel2);
312 int list2Size = list2.size();
313
314 assertNotSame("Ack lists should not be same", list1.toString(), list2.toString());
315 assertEquals("list2 should be shorter by one index", list1Size - 1, list2Size);
316 assertFalse("list2 should not contain the deleted ack", list2.containsKey(deletedAck.getId()));
317 }
