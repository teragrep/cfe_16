58public class TokenManagerTests {
59
60 TokenManager manager = new TokenManager();
61
62 /*
63 * Tests TokenManager's tokenIsMissing() method which checks if
64 * HttpServletRequest's header has an authentication token in it.
65 */
66 @Test
67 public void tokenCheckingTest() {
68
69 MockHttpServletRequest requestWithHttpHeaderAuth = new MockHttpServletRequest();
70 MockHttpServletRequest requestWithBasicAuth = new MockHttpServletRequest();
71 MockHttpServletRequest requestWithoutAuth = new MockHttpServletRequest();
72 String authToken = "AUTH_TOKEN_11111";
73
74 requestWithHttpHeaderAuth.addHeader("Authorization", authToken);
75 requestWithBasicAuth.addHeader("Authorization", "Basic x:" + authToken);
76
77 assertFalse("Token should be found from the request", manager.tokenIsMissing(requestWithHttpHeaderAuth));
78 assertFalse("Token should be found from the request", manager.tokenIsMissing(requestWithBasicAuth));
79 assertTrue("Token should not be found from the request", manager.tokenIsMissing(requestWithoutAuth));
80 }
81
82 /*
83 * Tests TokenManager's isTokenInBasic() method which checks if the
84 * HttpServletRequest's header's authentication token is in basic authentication
85 * format.
86 */
87 @Test
88 public void basicAuthCheckingTest() {
89 String authToken = "AUTH_TOKEN_11111";
90 String basicAuthHeader = "Basic x:" + authToken;
91
92 assertTrue("Authorization header should be in basic format", manager.isTokenInBasic(basicAuthHeader));
93 assertFalse(
94 "Authorization should not be in basic format when querying with only the authentication token.",
95 manager.isTokenInBasic(authToken)
96 );
97 }
98
99 /*
100 * Tests TokenManager's getTokenFromBasic() method which returns the
101 * authentication token when it is given in basic authentication format.
102 */
103 @Test
104 public void getTokenFromBasicAuthTest() {
105 String authToken = "AUTH_TOKEN_11111";
106 String basicAuthCredentials = "x:" + authToken;
107 String credentialsEncoded = Base64.getEncoder().encodeToString(basicAuthCredentials.getBytes());
108 String basicAuthHeader = "Basic " + credentialsEncoded;
109
110 assertEquals(
111 "Method should return the authentication token extracted from the Basic Authentication format",
112 authToken, manager.getTokenFromBasic(basicAuthHeader)
113 );
114 }
